Drawing a circle

// Create a workspace

svg = d3.select('#visualisation')
    .append('svg')
    .attr('width', '100%')
    .attr('height', '100%');

// Add a circle
// We are navigating the DOM just like jquery

svg.append('circle')
    .style('fill', 'white')
    .attr('r', 40)
    .attr('cx', 50)
    .attr('cy', 50)
    .on('mouseover', function() {
        d3.select(this).style('fill', 'aliceblue');
    })
    .on('mouseout', function() {
        d3.select(this).style('fill', 'white');
    });

.on('click', animate);

// d3 gives us helpful stuff, like animations!

function animate() {
    d3.select(this).transition()
        .delay(100)
        .duration(1000)
        .attr('r', 10)
        .attr('cx', 30)
        .style('fill', 'black');
}

Drawing data

// Create a workspace

svg = d3.select('#visualisation')
    .append('svg')
    .attr('width', '100%')
    .attr('height', '100%');

// Loading data
d3.csv('data/titanic.csv', function( data ) {
    // `data` is the data, already parsed
});

svg.selectAll( 'circle' )
    .data( data )
    .enter().append('circle')
    .style('fill', 'black')
    .style('opacity', 0.5)
    .attr('r', 20)
    .attr('cx', 50)
    .attr('cy', '50%');

// svg is <svg>
// we select all the circles in it
// of which there are none,
// and assign data to this selection
// then we append a circle for each
// row of data

var border = 50;

// DOM IS THE DATA

svg.selectAll('circle')
    .attr('cx', function(d) { return border + d.age*15; })
    .attr('cy', function(d) { return border + d.pclass*100; })
    .style('fill', function( d ) { return ( d.survived === '1' ? 'blue' : 'red' ); });
